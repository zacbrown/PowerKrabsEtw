Import-Module .\PowerKrabsEtw
$trace = New-EtwUserTrace
$provider = New-EtwUserProvider -ProviderName "Microsoft-Windows-WMI-Activity"
$filter = New-EtwCallbackFilter -EventId 11
Set-EtwCallbackFilter -UserProvider $provider -Filter $filter
Set-EtwUserProvider -Trace $trace -Provider $provider
cls
Start-EtwUserTrace -Trace $trace | Where-Object { $_.Operation -match "Win32_Process::Create" }
